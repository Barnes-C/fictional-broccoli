import setColor from './setColor';

/**
 * Funktion die auf der Ansichtsseite einer Anzeige beim dr체cken des "Anzeigen pr체fen" Knopfes ausgef체hrt wird. Es wird die Id der Anzeige ausgelesen und mit dieser dann ein Request auf den Analyze-Service gemacht
 *
 */
const sendData = () => {
  const id = parseInt(
    document
      .getElementById('viewad-extra-info')
      .lastChild.previousSibling.textContent.split(' ', 2)[1],
    10,
  );

  fetch(`http://localhost:4200/api/analyze/${id}`)
    .then((response) => response.json())
    .then((data) => {
      const button = document.getElementById('fraudAdButton');
      button.disabled = true;

      const score = data.fraud_score;
      setColor(score, button, 'button');
    });
};

export default sendData;
